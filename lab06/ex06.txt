双路反相PWM交替渐变LED工程，存放源码、双LED硬件接线说明、波形效果记录。
#define LED_A 2
#define LED_B 13
int val = 0;
int step = 1;

void setup() {
  pinMode(LED_A, OUTPUT);
  pinMode(LED_B, OUTPUT);
}

void loop() {
  analogWrite(LED_A, val);
  analogWrite(LED_B, 255 - val);
  val += step;
  if (val >= 255 || val <= 0) step = -step;
  delay(5);
}
